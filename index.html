// ALGORITMO MEJORADO: LECTURA DE PANTALLA ABSOLUTA Y OCULTACIÓN DE ZONAS GENÉRICAS function updateMapLabels() { if(!mapLabels.length) return; // 1. UMBRAL DE ZOOM const MIN_ZOOM_THRESHOLD = 2.0; if (scale < MIN_ZOOM_THRESHOLD) { mapLabels.forEach(lbl => lbl.textNode.style.opacity = '0'); return; } const targetScreenFontSize = 11; const fSize = targetScreenFontSize / scale; // FASE 1: Escritura DOM (Preparar textos y tamaños sin forzar recalcular la pantalla) mapLabels.forEach(lbl => { let name = dbData.municipalities[lbl.id]?.name; // CORRECCIÓN "ZONA XXX": Si no tiene nombre guardado y es un ID automático, lo ignoramos por completo if (!name && lbl.id.startsWith('zona_')) { lbl.skip = true; lbl.textNode.style.opacity = '0'; return; } lbl.skip = false; // Si tiene un ID real en el SVG (ej. id="Salamanca") pero no está en la DB, lo usamos if (!name) name = lbl.id.replace(/_/g, ' ').toUpperCase(); if(lbl.textNode.textContent !== name) lbl.textNode.textContent = name; lbl.textNode.setAttribute('font-size', `${fSize}px`); }); const vW = viewport.clientWidth; const vH = viewport.clientHeight; const viewportRect = viewport.getBoundingClientRect(); const occupied = []; // FASE 2: Lectura DOM (Bounding Rects absolutos) y Culling/Colisiones mapLabels.forEach(lbl => { if (lbl.skip) return; // CORRECCIÓN DESALINEACIÓN: Obtenemos la posición EXACTA en la que el navegador está pintando el texto const rect = lbl.textNode.getBoundingClientRect(); // Traducimos a coordenadas relativas de tu contenedor visual const screenX = rect.left - viewportRect.left + rect.width / 2; const screenY = rect.top - viewportRect.top + rect.height / 2; // Frustum Culling (Ocultar si sale de la pantalla visible) const padding = 20; if (screenX < -padding || screenX > vW + padding || screenY < -padding || screenY > vH + padding) { lbl.textNode.style.opacity = '0'; return; } // Colisión basada en las cajas de texto reales en píxeles de pantalla const margin = 2; // Margen de respiración entre textos const r = { l: rect.left, r: rect.right, t: rect.top, b: rect.bottom }; let overlap = occupied.some(o => (r.l - margin) < o.r && (r.r + margin) > o.l && (r.t - margin) < o.b && (r.b + margin) > o.t ); if(!overlap) { lbl.textNode.style.opacity = '1'; occupied.push(r); } else { lbl.textNode.style.opacity = '0'; } }); }
